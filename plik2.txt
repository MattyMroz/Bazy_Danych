CREATE OR ALTER PROCEDURE dbo.Zadanie3
AS
WITH KlienciOracle AS (
    SELECT * FROM OPENQUERY(ORACLE, 'SELECT CUSTOMERID, COUNTRY FROM CUSTOMERS')
),
KategorieOracle AS (
    SELECT * FROM OPENQUERY(ORACLE, 'SELECT CATEGORYID, CATEGORYNAME FROM CATEGORIES')
),
Dane1997 AS (
    SELECT 
        p.CategoryID, 
        SUM(od.UnitPrice * od.Quantity) AS Przychody
    FROM Northwind.dbo.Orders o
    JOIN Northwind.dbo.[Order Details] od ON o.OrderID = od.OrderID
    JOIN Northwind.dbo.Products p ON od.ProductID = p.ProductID
    JOIN Northwind.dbo.Shippers s ON o.ShipVia = s.ShipperID
    JOIN KlienciOracle c ON o.CustomerID = c.CUSTOMERID
    WHERE YEAR(o.OrderDate) = 1997
      AND c.COUNTRY = s.Country 
    GROUP BY p.CategoryID
),
Dane1998 AS (
    SELECT 
        p.CategoryID, 
        SUM(od.UnitPrice * od.Quantity) AS Przychody
    FROM Northwind.dbo.Orders o
    JOIN Northwind.dbo.[Order Details] od ON o.OrderID = od.OrderID
    JOIN Northwind.dbo.Products p ON od.ProductID = p.ProductID
    JOIN Northwind.dbo.Shippers s ON o.ShipVia = s.ShipperID
    JOIN KlienciOracle c ON o.CustomerID = c.CUSTOMERID
    WHERE YEAR(o.OrderDate) = 1998
      AND c.COUNTRY = s.Country
    GROUP BY p.CategoryID
)
SELECT TOP 1 
    ko.CATEGORYNAME,
    d97.Przychody AS Przychody1997,
    ISNULL(d98.Przychody, 0) AS Przychody1998,
    ((d97.Przychody - ISNULL(d98.Przychody, 0)) / d97.Przychody) * 100 AS ProcentowySpadek
FROM Dane1997 d97
LEFT JOIN Dane1998 d98 ON d97.CategoryID = d98.CategoryID
JOIN KategorieOracle ko ON d97.CategoryID = ko.CATEGORYID
WHERE d97.Przychody > ISNULL(d98.Przychody, 0)
ORDER BY ProcentowySpadek DESC;
GO

-- Przykłady wyzwalania procedury:
EXEC dbo.Zadanie3;
EXECUTE dbo.Zadanie3;
GO
