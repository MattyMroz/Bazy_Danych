-- USER SQL

DEFAULT TABLESPACE "USERS"
TEMPORARY TABLESPACE "TEMP";

-- ROLES
GRANT "CONNECT" TO TEST;
GRANT "RESOURCE" TO TEST;

-- SYSTEM PRIVILEGES
